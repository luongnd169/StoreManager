CREATE TABLE item (
	itemId INT PRIMARY KEY auto_increment,
	NAME nvarchar (50),
	type VARCHAR (20),
	price DOUBLE,
	quantity INT
);

CREATE TABLE itemdetail (
	id INT PRIMARY KEY auto_increment,
	itemId int not null,
	importPrice varchar(20),
	imei VARCHAR (20),
	provider int,
	importDate date,
	status boolean,
	exportDate date,
	customer int
);

create table salebill(billId int primary key auto_increment, billNo int not NULL, totalPrice varchar(20), customerPhone varchar(11), date DATE);
create table salebilldetail(bdId int primary key auto_increment, billNo int not NULL, name nvarchar(50), price varchar(20), quantity int, date DATE);
create table fee(feeId int primary key auto_increment, name nvarchar(100), value varchar(20), type boolean, date date);
create table customer(customerId int primary key auto_increment, name nvarchar(50), phone varchar(12), address nvarchar(100));
